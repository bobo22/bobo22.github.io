还记得大概两年前出去有位面试官问我改变this指向有几种方法，当时只回答了call和apply，后来才知道还有一个bind的存在，这边就简单介绍一下这三个东西，参考于MDN，并加入了一些自己的理解。可以移步MDN官网查看MDN

# call

首先我们要明白，call和apply语法合作用基本是一致的，不同的是两者调用方式不同。call调用为

function.call(thisArg, arg1, arg2, ...)
1

其中function为功能函数,thisArg为函数运行时的this指向。当然如果处于非严格模式下，传入null和undefined可以自动指向为全局对象。后面的即为参数列表。该方法是没有返回值的，会返回function功能函数的返回值。

下面我们开始讲用法。下面我们使用call实现一个继承功能。

function Product(name, price) {
  this.name = name;
  this.price = price;
}

function Food(name, price) {
  Product.call(this, name, price);
  this.category = 'food';
}

function Toy(name, price) {
  Product.call(this, name, price);
  this.category = 'toy';
}

var cheese = new Food('feta', 5);
var fun = new Toy('robot', 40);

第二种，使用call调用匿名函数

var animals = [
  { species: 'Lion', name: 'King' },
  { species: 'Whale', name: 'Fail' }
];

for (var i = 0; i < animals.length; i++) {
  (function(i) {
    this.print = function() {
      console.log('#' + i + ' ' + this.species
                  + ': ' + this.name);
    }
    this.print();
  }).call(animals[i], i);
}

第三种，我们使用call改变函数this的指向，同理，这边放一段代码

var obj = {
  animal: 'cats', sleepDuration: '12 and 16 hours'

greet.call(obj);  // cats typically sleep between 12 and 16 hours

# apply

apply方法，调用方式和call类似，不过除了给定一个代表this指向的参数之外，后面的传参是一个数组或者类数组

func.apply(thisArg, [argsArray])
1

下面举例，通过apply将数组元素添加至另一个数组

var array = ['a', 'b'];
var elements = [0, 1, 2];
array.push.apply(array, elements);
console.info(array); // ["a", "b", 0, 1, 2]

为什么这边可以传数组作为参数呢，如果是平时的array的push方法，我们只能push元素，如果push数组的话，会直接push进去，会生成一个二维数组，通过这种方法，就可以这种操作

# bind

bind方法，会创建一个新的函数。在 bind() 被调用时，这个新函数的 this 被指定为 bind() 的第一个参数，而其余参数将作为新函数的参数，供调用时使用

function.bind(thisArg[, arg1[, arg2[, ...]]])
1

用法，最简单的就如上面介绍的一样，创建一个函数。

this.x = 9;    // 在浏览器中，this 指向全局的 "window" 对象
var module = {
  x: 81,
  getX: function() { return this.x; }
};

module.getX(); // 81

var retrieveX = module.getX;
retrieveX();   
// 返回 9 - 因为函数是在全局作用域中调用的

// 创建一个新函数，把 'this' 绑定到 module 对象
// 新手可能会将全局变量 x 与 module 的属性 x 混淆
var boundGetX = retrieveX.bind(module);
boundGetX(); // 81

第一个输出81就不讲了，第二个输出9因为此时this指向的是他的调用者window，而下面一个，我们重新将retrieveX这个函数绑到了module上面，因此还会输出module里面的x，即为81.

第二点，官方解释叫偏函数，bind() 的另一个最简单的用法是使一个函数拥有预设的初始参数。只要将这些参数（如果有的话）作为 bind() 的参数写在 this 后面。当绑定函数被调用时，这些参数会被插入到目标函数的参数列表的开始位置，传递给绑定函数的参数会跟在它们后面。

function list() {
  return Array.prototype.slice.call(arguments);
}

function addArguments(arg1, arg2) {
    return arg1 + arg2
}

var list1 = list(1, 2, 3); // [1, 2, 3]

var result1 = addArguments(1, 2); // 3

// 创建一个函数，它拥有预设参数列表。
var leadingThirtysevenList = list.bind(null, 37);

// 创建一个函数，它拥有预设的第一个参数
var addThirtySeven = addArguments.bind(null, 37); 

var list2 = leadingThirtysevenList(); 
// [37]

var list3 = leadingThirtysevenList(1, 2, 3); 
// [37, 1, 2, 3]

var result2 = addThirtySeven(5); 
// 37 + 5 = 42 

var result3 = addThirtySeven(5, 10);
// 37 + 5 = 42 ，第二个参数被忽略

使用 window.setTimeout() 时，this 关键字会指向 window （或 global）对象。当类的方法中需要 this 指向类的实例时，你可能需要显式地把 this 绑定到回调函数，就不会丢失该实例的引用。 使用bind可以防止在定时器内部的this指向变成window。

function LateBloomer() {
  this.petalCount = Math.ceil(Math.random() * 12) + 1;
}

// 在 1 秒钟后声明 bloom
LateBloomer.prototype.bloom = function() {
  window.setTimeout(this.declare.bind(this), 1000);
};

LateBloomer.prototype.declare = function() {
  console.log('I am a beautiful flower with ' +
    this.petalCount + ' petals!');
};

var flower = new LateBloomer();
flower.bloom();  // 一秒钟后, 调用 'declare' 方法
